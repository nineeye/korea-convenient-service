(앞의 코드는 그대로 두고, 스크립트 하단에 추가) ... // 1초마다 현재 시간과 등록된 알림들을 비교 setInterval(() => { const now = new Date(); const nowTime = `${String(now.getHours()).padStart(2, '0')}:${String(now.getMinutes()).padStart(2, '0')}`; // 등록된 알림들을 하나씩 확인 document.querySelectorAll('.alarm-item').forEach(item => { // item.dataset.time에는 "19:21" 같은 형식이 저장되어야 함 if (item.dataset.time === nowTime && !item.classList.contains('fired')) { triggerAlarm(); // 알림 울림! item.classList.add('fired'); // 중복 울림 방지 alert("🔔 알림 시간입니다: " + item.dataset.name); } }); }, 1000); function addAlarm() { const name = document.getElementById('alarmName').value; const time = document.getElementById('alarmTime').value; // 이 값은 항상 "HH:MM" 형태임 if(!name || !time) return alert("내용과 시간을 입력해주세요!"); const div = document.createElement('div'); div.className = "alarm-item"; div.dataset.time = time; // 비교용 시간 저장 div.dataset.name = name; div.innerHTML = `

🔔 ${name} - ${time}

`; document.getElementById('alarmList').appendChild(div); }
